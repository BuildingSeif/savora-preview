### JS REPLACE: function renderCookStep(){
function renderCookStep(){
  try{
    const steps=getCookSteps();
    if(!steps||steps.length===0)return;
    const s=steps[cookStep];
    const r=RECIPES[currentRecipeIndex];
    if(!s||!r)return;
    
    /* Recipe title: .km-title OR .cook-tt */
    const titleEl=document.querySelector('#s4 .km-title')||document.querySelector('#s4 .cook-tt');
    if(titleEl)titleEl.textContent=r.title;
    
    /* Step label: #kmStepLabel OR #cookStepLabel */
    const stepLabel=document.getElementById('kmStepLabel')||document.getElementById('cookStepLabel');
    if(stepLabel)stepLabel.innerHTML='Étape '+s.n+' sur '+steps.length;
    
    /* Step number watermark: #kmWatermark */
    const watermark=document.getElementById('kmWatermark');
    if(watermark)watermark.textContent=s.n;
    
    /* Instruction text: #kmInstruction OR #cookInst */
    const instEl=document.getElementById('kmInstruction')||document.getElementById('cookInst');
    if(instEl)instEl.innerHTML=s.text;
    
    /* Step time badge: #kmStepTime */
    const stepTime=document.getElementById('kmStepTime');
    if(stepTime)stepTime.textContent=s.time;
    
    /* Hint/Phase: #cookStepHint (legacy) */
    const hintEl=document.getElementById('cookStepHint');
    if(hintEl)hintEl.textContent=cookStep===0?'PRÉPARATION':cookStep===steps.length-1?'FINITION':'CUISSON';
    
    /* Recipe image: #kmRecipeImg OR #cookImg */
    const cookImg=document.getElementById('kmRecipeImg')||document.getElementById('cookImg');
    if(cookImg)cookImg.src=s.img;
    
    /* Timer value: #kmTimerValue OR #timerD */
    const timerVal=document.getElementById('kmTimerValue')||document.getElementById('timerD');
    if(timerVal)timerVal.textContent=s.time;
    
    /* Initialize timer state */
    timerV=s.timeS;
    clearInterval(timerInt);
    timerOn=false;
    
    /* Timer icon: #timerI (legacy) */
    const timerIcon=document.getElementById('timerI');
    if(timerIcon)timerIcon.textContent='play_arrow';
    
    /* Progress bar: #kmProgress OR #cookProgress */
    const cp=document.getElementById('kmProgress')||document.getElementById('cookProgress');
    if(cp)cp.style.width=((cookStep+1)/steps.length*100)+'%';
    
    /* Previous button: #kmPrevBtn OR #cookPrevBtn */
    const prevBtn=document.getElementById('kmPrevBtn')||document.getElementById('cookPrevBtn');
    if(prevBtn){
      prevBtn.disabled=cookStep===0;
      prevBtn.style.opacity=cookStep===0?'.4':'1';
    }
    
    /* Next button: #kmNextBtn OR #cookNextBtn */
    const nextBtn=document.getElementById('kmNextBtn')||document.getElementById('cookNextBtn');
    if(nextBtn){
      if(cookStep===steps.length-1){
        nextBtn.innerHTML='Terminer <span class="ms">check</span>';
      }else{
        nextBtn.innerHTML='Suivant <span class="ms">arrow_forward</span>';
      }
    }
    
    /* Populate ingredient drawer */
    const ingList=document.getElementById('kmIngList');
    if(ingList&&r.ingredients){
      ingList.innerHTML='';
      r.ingredients.forEach(function(ing){
        if(ing&&ing.name&&ing.qty){
          ingList.innerHTML+='<div class="km-ing-item" onclick="kmCheckIng(this)"><div class="km-ing-check"><span class="ms">check</span></div><div class="km-ing-name">'+ing.name+'</div><div class="km-ing-qty">'+ing.qty+'</div></div>';
        }
      });
    }
  }catch(e){
    console.warn('renderCookStep error:',e);
  }
}
